Konwencje wspólne (obowiązują we wszystkich sekcjach)

Cztery rozstrzygnięcia zdejmują dwuznaczność z redline'ów. Reszta worksheetu je zakłada.

1 · px = pt @1x. Redline w CSS px czytamy jako pt 1:1 — teza 24 px → 24 pt, szyna 236 px → 236 pt. Retina @2x skaluje automatycznie; ręcznie tylko hairline'y (#5).
2 · Wagi 400 / 500 / 700 → .regular / .medium / .bold. „500" = .medium (nie .semibold). 600/.semibold w projekcie nie występuje — nie wprowadzać.
3 · Fonty → SF systemowe. Neue Montreal → SF Pro (Display), Neue Haas → SF Pro Text, mono → SF Mono. NSFont.systemFont sam przełącza optyk Display/Text na progu 20 pt = nasz próg. Diakrytyki PL pełne.

extension NSAttributedString {
  static func ts(_ t:String, pt:CGFloat, weight:NSFont.Weight = .regular,
                 em:CGFloat = 0, mult:CGFloat = 1.0, color:NSColor,
                 upper:Bool = false, mono:Bool = false) -> NSAttributedString {
    let f = mono ? TS.mono(pt,weight) : TS.font(pt,weight)
    let p = NSMutableParagraphStyle(); p.lineHeightMultiple = mult
    return NSAttributedString(string: upper ? t.uppercased() : t,
      attributes: [.font:f, .foregroundColor:color, .paragraphStyle:p, .kern: em*pt])
  }
}

enum Ease {
  static let standard   = CAMediaTimingFunction(controlPoints: 0.4,0, 0.2,1)
  static let decelerate = CAMediaTimingFunction(controlPoints: 0,  0, 0.2,1)
  static let accelerate = CAMediaTimingFunction(controlPoints: 0.4,0, 1,  1)
}
func slideIn(_ l:CALayer, dy:CGFloat, dur:CFTimeInterval, timing:CAMediaTimingFunction) {
  if NSWorkspace.shared.accessibilityDisplayShouldReduceMotion { l.opacity = 1; return }  // cięcie
  let o = CABasicAnimation(keyPath:"opacity"); o.fromValue = 0; o.toValue = 1
  let t = CABasicAnimation(keyPath:"transform.translation.y"); t.fromValue = dy; t.toValue = 0
  let g = CAAnimationGroup(); g.animations = [o,t]; g.duration = dur
  g.timingFunction = timing; g.fillMode = .backwards
  l.add(g, forKey:"slideIn"); l.opacity = 1
}
